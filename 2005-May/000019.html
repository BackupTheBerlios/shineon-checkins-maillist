begin
  with T:TStringlist := new TStringlist do
  begin
    T.Add('AAA');
    T.Add('BBB');
    T.Delete(T.IndexOf('AAA'));
    NUnit.Framework.Assert.AreEqual(1, T.Count, 'TStringlist.Count is wrong');
  end;
end;

begin
  with T:TStringlist := new TStringlist do
  begin
    T.Add('BBB');
    T.Add('AAA');
    T.Delete(0);

  end;
end;
